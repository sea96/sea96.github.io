题意

一个人要读$n$个topic，每天一个topic。在第$k$天读第$x$个topic得到的level是$k*w[x]$，现在有$m$个约束，即读topic的顺序，问得到的最大的level是多少。

数据

$1 \leq n \leq 18$
$ 1 \leq w \leq 100$

题解

很容易想到是拓扑排序，如果使用贪心策略每次取出$w$最小的，用优先队列实现，但是这个是错误的，因为只保证了当前的最优解，没有保证全局最优解。
所以考虑dp的方法，注意到$n$的数据范围很小，自然想到用状态压缩dp，状态转移的条件和拓扑排序一样，入度为0的时候才能转移。

代码

#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N = 20;
map<string, int> ID;
struct Edge {
    int u, v;
};
vector<Edge> edges;
int dp[(1<<18)+5];
bool vis[N];
int w[N];
int deg[N];
int n, m;

int solve() {
    memset(dp, 0, sizeof(dp));
    for (int mask=0; mask<(1<<n); ++mask) {
        memset(vis, false, sizeof(vis));
        for (int i=0; i<n; ++i) if (mask & (1<<i)) vis[i] = true;
        int day = __builtin_popcount(mask);
        bool flag = true;
        memset(deg, 0, sizeof(deg));
        for (auto e: edges) {
            if (!vis[e.u] && vis[e.v]) { flag = false; break; }
            if (!vis[e.u]) deg[e.v]++;
        }
        if (!flag) continue;
        for (int i=0; i<n; ++i) {
            if (!vis[i] && !deg[i]) {
                dp[mask|(1<<i)] = max(dp[mask|(1<<i)], dp[mask]+(day+1)*w[i]);
            }
        }
    }
    return dp[(1<<n)-1];
}

int main() {
    int T;
    cin >> T;
    while (T--) {
        cin >> n >> m;
        edges.clear();
        ID.clear();
        string topic, topic2, tmp;
        getline(cin, tmp);
        for (int i=0; i<n; ++i) {
            w[i] = 0; topic = "";
            getline(cin, tmp);
            for (int j=0; j<tmp.length(); ++j) {
                if (tmp[j] >= '0' && tmp[j] <= '9') {
                    w[i] = w[i]*10 + (tmp[j]-'0');
                } else topic += tmp[j];
            }
            ID[topic] = i;
        }
        edges.clear();
        for (int i=1; i<=m; ++i) {
            getline(cin, tmp);
            topic = topic2 = "";
            bool fir = true;
            for (int j=0; j<tmp.length(); ++j) {
                if (tmp[j] == '-') {
                    j += 3;
                    fir = false;
                } else {
                    if (fir) topic += tmp[j];
                    else topic2 += tmp[j];
                }
            }
            topic2 += ' ';
            edges.push_back(Edge{ID[topic], ID[topic2]});
        }
        cout << solve() << '\n';
    }
    return 0;
}
